list_a=[1,2,3,4,5,11,14,6]
list_b=[6,7,8,9,10,2,3,1]
length_a=len(list_a)
length_b=len(list_b)
if(length_b==length_a):
    print("List are in same length")
else:
    print("list are not in same length")
total_a=sum(list_a)
total_b=sum(list_b)
if(total_b==total_a):
    print("list sum are equal")
else:
    print("list sum are not equal")
output=any(check in list_b for check in list_a)
print("common value:",output)
print(set(list_b).intersection(set(list_a)))
