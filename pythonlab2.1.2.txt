n1=0
n2=1
count=0
limit=int(input("enter the limit:"))
if limit<=0:
     print("enter a positive number:")
elif limit==1:
     print("fibonacci series upto",limit)
     print(n1)
else:
    print("fibonacci series:")
    while count<limit:
        print(n1)
        n3=n1+n2
        n1=n2
        n2=n3
        count=count+1
